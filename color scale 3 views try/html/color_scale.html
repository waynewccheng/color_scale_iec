% convert to CIELAB
lab = rgb2lab(c/255,'ColorSpace','srgb');

lch = lab;
for i = 1:8
    a = lab(i,2);
    b = lab(i,3);
    rad = atan2(b,a);
    chrm = (a.^2 + b.^2) .^ 0.5;
    lch(i,3) = rad * 180 / pi;
    lch(i,2) = chrm;
end

% dE
dE_allpair = zeros(8,8);
for i = 1:8
    for j = 1:8
        lab1 = lab(i,:);
        lab2 = lab(j,:);
        dE = sum((lab1-lab2).^2).^0.5;
        dE_allpair(i,j) = dE;
    end
end
